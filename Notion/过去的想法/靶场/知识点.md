# 工具
## nmap

nmap (-sS -sT -sU) -P -sV -O
nmap -sS -sV -O -p- 目标IP SYN、服务识别、操作系统识别、扫描所有端口
-sn 主机发现
--script=vuln IP 脚本

## Feroxbuster
Feroxbuster -u IP -w .txt -x php -o
feroxbuster -u http://IP/ --depth=1 -x php,html,txt,zip,bak

## ffuf
ffuf -c -w /usr/share/seclists/Discovery/Web-Content/burp-parameter-names.txt -u 'http://IP/?FUZZ=1' -fw 100
### 目录
>ffuf -w /usr/share/seclists/Discovery/Web-Content/common.txt \
-u http://target/FUZZ
### 参数
>ffuf -w burp-parameter-names.txt \
-u http://target/?FUZZ=test
### 子域名
>ffuf -w subdomains.txt \
-u http://FUZZ.target.com

## tty
```
python -c 'import pty; pty.spawn("/bin/sh")'
Ctl+z #退出当前shell
stty raw -echo;fg #重新进入shell
reset
xterm
stty rows 24 columns 160
export TERM=xterm
```
